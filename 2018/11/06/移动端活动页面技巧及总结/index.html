function moveTop ($obj) {
            var $child=$obj.children().clone();
            $obj.append($child);
            var initTop=0;
            var timer=null;
            clearInterval(timer);
            timer=setInterval(function () {
                    initTop=$obj.scrollTop();
                    initTop++;
                    if(initTop>=$child.height()){
                            initTop-=$child.height();
                    }
                    $obj.scrollTop(initTop);
            },50);
    }
    moveTop($(".content-box"));
